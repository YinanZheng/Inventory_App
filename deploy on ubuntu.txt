run_as shiny;

server {
  listen 3838;

  # 配置 erp-cn 应用

    site_dir /srv/shiny-server/erp-cn;
    log_dir /var/log/shiny-server;
    directory_index on;
  }

  # 配置 erp-us 应用
  location /erp-us {
    site_dir /srv/shiny-server/erp-us;
    log_dir /var/log/shiny-server;
    directory_index on;
  }

  # 配置 erp-mobile 应用
  location /erp-mobile {
    site_dir /srv/shiny-server/erp-mobile;
    log_dir /var/log/shiny-server;
    directory_index on;
  }
}
